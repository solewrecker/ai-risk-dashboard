## 💻 TECHNICAL IMPLEMENTATION

### Database Schema
```sql
-- Enhanced users table
ALTER TABLE users ADD COLUMN user_stats JSONB DEFAULT '{
  "assessments_completed": 0,
  "tools_compared": 0,
  "team_invites_sent": 0,
  "unlocked_themes": ["default"],
  "current_theme": "default",
  "achievements": []
}';

-- Achievements tracking
CREATE TABLE user_achievements (
  id SERIAL PRIMARY KEY,
  user_id INTEGER REFERENCES users(id),
  achievement_key VARCHAR(100),
  unlocked_at TIMESTAMP DEFAULT NOW(),
  milestone_value INTEGER
);
```

### Theme System Implementation
```javascript
// Theme management
const ThemeManager = {
  themes: {
    default: { name: "Default", unlockLevel: 0 },
    darkCoder: { name: "Dark Coder", unlockLevel: 1 },
    matrix: { name: "Matrix", unlockLevel: 10 },
    cyberpunk: { name: "Cyberpunk", unlockLevel: 50 },
    corporate: { name: "Corporate Pro", unlockLevel: 100 }
  },

  applyTheme: (themeName) => {
    document.documentElement.setAttribute('data-theme', themeName);
    localStorage.setItem('selectedTheme', themeName);
  },

  checkUnlocks: (userStats, subscription) => {
    if (subscription === 'pro') return Object.keys(this.themes);
    
    return Object.entries(this.themes)
      .filter(([key, theme]) => userStats.assessments_completed >= theme.unlockLevel)
      .map(([key]) => key);
  }
};

// Achievement unlock modal
const showUnlockModal = (achievement) => {
  const modal = `
    <div class="unlock-modal">
      <div class="confetti-animation"></div>
      <h2>🎉 Achievement Unlocked!</h2>
      <div class="achievement-icon">${achievement.icon}</div>
      <h3>${achievement.reward}</h3>
      <div class="theme-preview" data-theme="${achievement.themeKey}"></div>
      <button onclick="applyTheme('${achievement.themeKey}')">Apply Now</button>
      <button onclick="previewTheme('${achievement.themeKey}')">Preview First</button>
    </div>
  `;
  document.body.insertAdjacentHTML('beforeend', modal);
};
```

### Progress Tracking
```javascript
// Track user actions for gamification
const trackUserAction = async (action, userId) => {
  const actions = {
    'assessment_completed': () => incrementStat('assessments_completed'),
    'tool_compared': () => incrementStat('tools_compared'),
    'team_invite_sent': () => incrementStat('team_invites_sent')
  };

  if (actions[action]) {
    await actions[action]();
    await checkAndUnlockAchievements(userId);
  }
};

const checkAndUnlockAchievements = async (userId) => {
  const userStats = await getUserStats(userId);
  const newAchievements = [];

  Object.entries(achievements).forEach(([key, achievement]) => {
    const statValue = userStats[achievement.statKey] || 0;
    if (statValue >= achievement.threshold && !userStats.achievements.includes(key)) {
      newAchievements.push({ key, ...achievement });
    }
  });

  if (newAchievements.length > 0) {
    await saveNewAchievements(userId, newAchievements);
    showUnlockModal(newAchievements[0]);
  }
};
```
